document.querySelector(".merger-alt-btn").onclick = function()
{
    let input = document.querySelector("#input");
    let output = document.querySelector("#output");

    let str = input.value;

    str = str.replace(/:\s*$|^(?=\n)$|^\s*|\s*$|\n\n+/gm, '')
    


    console.log(str);

    let newStr = "";
    for(let i = 0 ; i < str.split('\n').length;i += 1)
    {
        if(i % 2== 0)
        { 
            let left = str.split('\n')[i];

            newStr += left + ' is ';
            newStr += str.split('\n')[i + 1];
            newStr += '\n'; 
        }
    }


    output.value = newStr;
}

document.querySelector(".merger-tab-btn").onclick = function()
{
    let input = document.querySelector("#input");
    let output = document.querySelector("#output");

    let str = input.value;
    console.log(str);


    newStr = "";
    let tabs = str.split('\n\n\n');

    let left = tabs[0].split('\n');
    let right = tabs[1].split('\n');

    for(let i = 0 ; i < left.length;i += 1)
    {
        newStr += left[i]+' is '+right[i];
        newStr+='\n';
    }


    output.value = newStr;
}
